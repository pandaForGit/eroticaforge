### 1. 新建文件：`NovelClassifier.java`

路径建议：`src/main/java/com/eroticaforge/application/service/NovelClassifier.java`

```java
package com.eroticaforge.application.service;

import com.fasterxml.jackson.databind.ObjectMapper;
import dev.langchain4j.model.chat.ChatModel;
import dev.langchain4j.model.output.Response;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * 小说合集批量分类工具
 * 用于对 10G 小说进行提前分类（现代/二次元/SM/NTR 等）
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class NovelClassifier {

    private final ChatModel summarizerModel;   // 推荐使用 4B 模型，速度快
    private final ObjectMapper objectMapper;

    private static final int MAX_CHARS_PER_FILE = 4500;  // 只分析前4500字符，足够判断类型

    /**
     * 批量分类整个文件夹下的小说
     * @param sourceFolder 源文件夹路径，例如 F:\novel_collection\
     * @param outputJsonPath 输出分类结果的 JSON 文件路径
     */
    public void classifyFolder(String sourceFolder, String outputJsonPath) throws IOException {
        File folder = new File(sourceFolder);
        if (!folder.exists() || !folder.isDirectory()) {
            throw new IllegalArgumentException("文件夹不存在: " + sourceFolder);
        }

        Map<String, ClassificationResult> results = new HashMap<>();
        AtomicInteger processed = new AtomicInteger(0);
        AtomicInteger success = new AtomicInteger(0);

        log.info("开始分类小说文件夹: {}", sourceFolder);

        Files.walk(folder.toPath())
                .filter(Files::isRegularFile)
                .filter(path -> path.toString().toLowerCase().endsWith(".txt"))
                .forEach(path -> {
                    try {
                        String fileName = path.getFileName().toString();
                        String content = readFirstNChars(path, MAX_CHARS_PER_FILE);

                        if (content.length() < 300) {
                            log.warn("文件内容过短，跳过: {}", fileName);
                            return;
                        }

                        ClassificationResult result = classifySingleNovel(fileName, content);
                        results.put(fileName, result);

                        success.incrementAndGet();
                        log.info("已分类 [{}/?] {}", success.get(), fileName);

                    } catch (Exception e) {
                        log.error("分类失败: {}", path.getFileName(), e);
                    }

                    processed.incrementAndGet();
                    if (processed.get() % 50 == 0) {
                        log.info("进度: 已处理 {} 个文件", processed.get());
                    }
                });

        // 保存分类结果
        objectMapper.writerWithDefaultPrettyPrinter().writeValue(new File(outputJsonPath), results);

        log.info("分类完成！共处理 {} 个文件，成功 {} 个，结果保存至: {}", 
                processed.get(), success.get(), outputJsonPath);
    }

    /**
     * 对单本小说进行分类
     */
    private ClassificationResult classifySingleNovel(String fileName, String content) {
        String prompt = """
            你是一个专业的小说分类助手。请对以下小说内容进行准确分类，只输出合法 JSON，不要任何其他文字。

            小说文件名: %s
            小说前4500字内容:
            %s

            请严格按照以下格式输出 JSON：
            {
              "title": "小说标题（从内容推断或使用文件名）",
              "main_type": "现代/二次元/都市/校园/古风/其他",
              "tags": ["NTR", "SM", "调教", "脑洗", "扶她", "乱伦", "多P", "纯爱", "重口", ...],
              "style": "细腻心理描写/重口味/对话驱动/二次元轻小说风/现实主义",
              "intensity": 1-5,
              "summary": "一句话剧情摘要（不超过30字）",
              "recommended_for": ["NTR故事", "SM调教故事", "二次元日常"等]
            }
            """.formatted(fileName, content);

        Response<String> response = summarizerModel.generate(prompt);

        try {
            return objectMapper.readValue(response.content(), ClassificationResult.class);
        } catch (Exception e) {
            log.warn("JSON 解析失败，使用默认值: {}", fileName);
            return new ClassificationResult(fileName, "未知", new String[]{}, "未知", 3, "解析失败", new String[]{});
        }
    }

    /**
     * 读取文件前 N 个字符
     */
    private String readFirstNChars(Path path, int maxChars) throws IOException {
        String content = Files.readString(path);
        return content.length() > maxChars ? content.substring(0, maxChars) : content;
    }

    /**
     * 分类结果实体类
     */
    public record ClassificationResult(
            String title,
            String main_type,
            String[] tags,
            String style,
            int intensity,
            String summary,
            String[] recommended_for
    ) {}
}
```
